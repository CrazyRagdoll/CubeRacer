97 inline const Vector3 slerp( float t, const Vector3 & unitVec0, const Vector3 & unitVec1 )
98 {
99  float recipSinAngle, scale0, scale1, cosAngle, angle;
100  cosAngle = dot( unitVec0, unitVec1 );
101  if ( cosAngle < _VECTORMATH_SLERP_TOL ) {
102  angle = acosf( cosAngle );
103  recipSinAngle = ( 1.0f / sinf( angle ) );
104  scale0 = ( sinf( ( ( 1.0f - t ) * angle ) ) * recipSinAngle );
105  scale1 = ( sinf( ( t * angle ) ) * recipSinAngle );
106  } else {
107  scale0 = ( 1.0f - t );
108  scale1 = t;
109  }
110  return ( ( unitVec0 * scale0 ) + ( unitVec1 * scale1 ) );
111 }

125 inline void loadHalfFloats( Vector3 & vec, const unsigned short * hfptr )
126 {
127  union Data32 {
128  unsigned int u32;
129  float f32;
130  };
131 
132  for (int i = 0; i < 3; i++) {
133  unsigned short fp16 = hfptr[i];
134  unsigned int sign = fp16 >> 15;
135  unsigned int exponent = (fp16 >> 10) & ((1 << 5) - 1);
136  unsigned int mantissa = fp16 & ((1 << 10) - 1);
137 
138  if (exponent == 0) {
139  // zero
140  mantissa = 0;
141 
142  } else if (exponent == 31) {
143  // infinity or nan -> infinity
144  exponent = 255;
145  mantissa = 0;
146 
147  } else {
148  exponent += 127 - 15;
149  mantissa <<= 13;
150  }
151 
152  Data32 d;
153  d.u32 = (sign << 31) | (exponent << 23) | mantissa;
154  vec[i] = d.f32;
155  }
156 }

159 {

161  unsigned int u32;
162  float f32;
163  };
164 
165  for (int i = 0; i < 3; i++) {
166  Data32 d;
167  d.f32 = vec[i];
168 
169  unsigned int sign = d.u32 >> 31;
170  unsigned int exponent = (d.u32 >> 23) & ((1 << 8) - 1);
171  unsigned int mantissa = d.u32 & ((1 << 23) - 1);;
172 
173  if (exponent == 0) {
174  // zero or denorm -> zero
175  mantissa = 0;
176 
177  } else if (exponent == 255 && mantissa != 0) {
178  // nan -> infinity
179  exponent = 31;
180  mantissa = 0;
181 
182  } else if (exponent >= 127 - 15 + 31) {
183  // overflow or infinity -> infinity

1102  unsigned int exponent = (fp16 >> 10) & ((1 << 5) - 1);
